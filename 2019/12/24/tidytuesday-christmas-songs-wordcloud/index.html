# TidyTuesday: WordCloud for Christmas Music Billboards

library(tidyverse)
library(tidytext)
data(stop_words)
library(wordcloud2)

tuesdata <- tidytuesdayR::tt_load("2019-12-24")
lyrics <- tuesdata$christmas_lyrics

lyc <- lyrics %>% unnest_tokens(word, lyric) %>%
    anti_join(stop_words) %>%
    count(word, sort = TRUE)

## Joining, by = "word"

wordcloud2(lyc)
